title Authentication & Recommandation Sequence

User->+Server: Authentication Request
note right of Server: Authentification
Server->-User: Authentication Response

User->+Server: Ajout_Theme Request
Server->+Recommandation: Ajout_Theme Request
note right of Recommandation : Recommandation 
Recommandation->+Theme: ListeTheme Request
note right of Theme : ListeTheme
Theme->-Recommandation : ListeTheme Response
Recommandation->Server: ListeTheme Response
alt theme exist
	Server->+Student: AddReferent Request
	note right of Student: AddReferent
	Student->-Server: AddReferent Response
else theme not exist 
	Server->+Theme: AddTheme Request
	note right of Theme : AddTheme 
	Theme->-Server: AddTheme Response
	Server->+Student: AddReferent Request
	note right of Student: AddReferent
	Student->-Server: AddReferent Response
end 
Recommandation->-Server: Ajout_Theme Response
Server->-User: Ajout_Theme Response

User->+Server: Utilisation_Courante Request
Server->+Theme: ListeTheme Request
note right of Theme : ListeTheme
Theme->-Server: ListeTheme Response
Server->User: ListeTheme Response

alt referent_seul
	User->+Server: Referent Request
	Server->+Student : Referent Request
	note right of Student : Referent
	Student->-Server: Referent Response
	Server->-User: Referent Response
else referent_liste
	User->+Server: Liste_Referent Request
	Server->+Student : Liste_Referent Request
	note right of Student : Liste_Referent 
	Student->-Server: Liste_Referent Response
	Server->-User: Liste_Referent Response
end

User->+Server: Liste_Referent Request
Server->+Student : Liste_Referent Request
note right of Student : Liste_Referent 
Student->-Server: Liste_Referent Response
Server->-User: Liste_Referent Response

alt ajouter_like
	User->+Server: Recommander_Referent Request
	Server->+Recommandation: Recommander_Referent Request
	note right of Recommandation: Recommander_Referent
	Recommandation->Recommandation: Add_Like
	note right of Recommandation : Add_Like
	Recommandation->-Server: Recommander_Referent Response
	Server->-User: Recommander_Referent Response
else retirer_like
	User->+Server: Retirer_Referent Request
	Server->+Recommandation: Retirer_Referent Request
	note right of Recommandation: Retirer_Referent 
	Recommandation->Recommandation: Remove_Like
	note right of Recommandation : Remove_Like
	Recommandation->-Server: Retirer_Referent Response
	Server->-User: Retirer_Referent Response
end

Server->-User: Utilisation_Courante Response
